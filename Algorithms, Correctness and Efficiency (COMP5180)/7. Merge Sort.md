function mergeSort(arr)

n = length(arr)

m = n/2

leftHalf = mergeSort(arr[0-m])

rightHalf = mergeSort(arr[m-n])

sortedArr = merge(leftHalf, rightHalf)

return sortedArr

end
```

```
function merge(a, b)

c = new array [length(a) + length(b)]

i = 0, j = 0, k = 0

while i < length(a) OR j < length(b)

  if a[i] < b[j]

    c[k] = a[i], i++, k++

  else

    c[k] = b[j], j++, k++

end
